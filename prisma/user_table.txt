CREATE TABLE public."User"
(
    id integer NOT NULL DEFAULT nextval('"User_id_seq"'::regclass),
    username character varying(256) COLLATE pg_catalog."default" NOT NULL,
    email character varying(256) COLLATE pg_catalog."default" NOT NULL,
    password character varying(256) COLLATE pg_catalog."default" NOT NULL,
    created character varying(16) COLLATE pg_catalog."default" NOT NULL,
    updated character varying(16) COLLATE pg_catalog."default" NOT NULL,
    CONSTRAINT "User_pkey" PRIMARY KEY (id),
    CONSTRAINT unique_constraint UNIQUE (email, username)
)
